cmake_minimum_required(VERSION 3.14.2)
project(aapt2-host)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 20)

# 平台相关编译选项
if(WIN32)
    # Windows (MinGW) 不需要 -fPIC
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wno-attributes")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wno-attributes")
    # 静态链接 libgcc/libstdc++，产物不依赖 MinGW 运行时
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -static")
else()
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fPIC -Wno-attributes")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fPIC -Wno-attributes")
endif()

set(HOST_BUILD ON)
set(SRC ${PROJECT_SOURCE_DIR}/submodules)

# GCC C++20 兼容性：全局注入 host_availability.h
if(NOT CMAKE_CXX_COMPILER_ID STREQUAL "AppleClang" AND NOT CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -include ${PROJECT_SOURCE_DIR}/cmake_host/host_availability.h")
endif()

add_definitions(-D_FILE_OFFSET_BITS=64)

# 通用 host 构建定义
add_definitions(
    -DFAKE_LOG_DEVICE=1
    -DSTATIC_ANDROIDFW_FOR_TOOLS
    -D_GNU_SOURCE
    -DNDEBUG
    -DNO_MEDIA_BACKEND
    -DNO_X_BACKEND
    -DNO_DB_BACKEND
)

# Windows 额外定义
if(WIN32)
    add_definitions(
        -DUSE_MINGW
        -D_WIN32_WINNT=0x0601
        -DNOMINMAX
        -DUNICODE
        -D_UNICODE
    )
endif()

# protobuf 配置
set(protobuf_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_PROTOC_BINARIES OFF CACHE BOOL "" FORCE)
set(protobuf_BUILD_LIBPROTOC ON CACHE BOOL "" FORCE)

# 第三方库
add_subdirectory(submodules/fmtlib EXCLUDE_FROM_ALL)
add_subdirectory(submodules/libpng EXCLUDE_FROM_ALL)
add_subdirectory(submodules/expat/expat EXCLUDE_FROM_ALL)
add_subdirectory(submodules/protobuf EXCLUDE_FROM_ALL)
add_subdirectory(submodules/googletest EXCLUDE_FROM_ALL)

find_package(ZLIB REQUIRED)

add_subdirectory(cmake_host)
